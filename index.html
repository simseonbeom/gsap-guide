Green Sock Animation Platform

gsap.to( ) 

gsap.to('.box-1', { duration: 1, x: 300, rotation:360 }) // duration 속성 명시 
gsap.to('.box-1', 1, {x: 300, rotation:360 }) // duration 속성 생략
duration 속성은 명시적으로 사용하길 권장합니다.

gsap.from( ) 

gsap.from('.box-2', { duration: 1, x: 300, rotation:360 })

gsap.fromTo( ) 

gsap.fromTo('.box-3',{ duration: 1, x: 300 ,rotation:360 },{ duration: 1, x: 0, rotation:0 })

gsap.set( ) 

gsap.set('.box-4', { x: 300, rotation:210 })
set 속성엔 duration 속성을 사용할 수 없습니다.

gsap.to( ) and gsap.set( ) 

gsap.to('.box-5', {duration:1, x: 300, rotation:360, onComplete:()=>{ gsap.set('.box-5',{x:0,rotation:0})} })

Timeline gsap.to( )


                let tl = gsap.timeline();

                tl.to('.box-6:nth-child(1)', { duration: 1, x: 300, rotation: 360 });
                tl.to('.box-6:nth-child(2)', { duration: 1, x: 300, rotation: 360 });
                tl.to('.box-6:nth-child(3)', { duration: 1, x: 300, rotation: 360 });
                tl.to('.box-6:nth-child(4)', { duration: 1, x: 300, rotation: 360 });
타임라인은 이전 트윈이 끝난 후 다음 트윈이 시작됩니다.

gsap.to( stagger )

gsap.to('.box-7',{x:300, rotation:360, stagger:0.1})
stagger값이 음수(-)인 경우 애니메이션이 거꾸로 시작합니다.
